CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(255) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL  -- Store the *hashed* password here
);

CREATE TABLE data (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(255) NOT NULL,
    data TEXT,  -- Or VARCHAR(255) if data is short
    FOREIGN KEY (username) REFERENCES users(username) -- Optional: Enforce referential integrity
);
